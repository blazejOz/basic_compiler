Program      ::= { Statement }

Statement    ::= VarDeclStmt
               | PrintStmt
               | ExprStmt

VarDeclStmt  ::= "int" IDENT "=" Expr ";"
PrintStmt    ::= "print" "(" Expr ")" ";"
ExprStmt     ::= Expr ";"

Expr         ::= Term { ("+" | "-") Term }
Term         ::= Factor { ("*" | "/") Factor }
Factor       ::= NUMBER  | IDENT | "(" Expr ")"
